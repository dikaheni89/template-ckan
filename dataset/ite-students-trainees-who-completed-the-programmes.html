Example: Javascript »

A simple ajax (JSONP) request to the data API using jQuery.

var data = {
  resource_id: '4e35400a-5a45-4483-9363-df01fa59584a', // the resource id
  limit: 5, // get 5 results
  q: 'jones' // query for 'jones'
};
$.ajax({
  url: 'https://data.gov.sg/api/action/datastore_search',
  data: data,
  dataType: 'jsonp',
  success: function(data) {
    alert('Total results found: ' + data.result.total)
  }
});
Example: Python » 
import urllib
url = 'https://data.gov.sg/api/action/datastore_search?resource_id=4e35400a-5a45-4483-9363-df01fa59584a&limit=5&q=title:jones'
fileobj = urllib.urlopen(url)
print fileobj.read()
