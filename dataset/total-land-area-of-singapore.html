Example: Javascript »

A simple ajax (JSONP) request to the data API using jQuery.

var data = {
  resource_id: 'f4bbfac9-c3ed-4f71-9b9a-238517b214ef', // the resource id
  limit: 5, // get 5 results
  q: 'jones' // query for 'jones'
};
$.ajax({
  url: 'https://data.gov.sg/api/action/datastore_search',
  data: data,
  dataType: 'jsonp',
  success: function(data) {
    alert('Total results found: ' + data.result.total)
  }
});
Example: Python » 
import urllib
url = 'https://data.gov.sg/api/action/datastore_search?resource_id=f4bbfac9-c3ed-4f71-9b9a-238517b214ef&limit=5&q=title:jones'
fileobj = urllib.urlopen(url)
print fileobj.read()
